Question 3: Q3-Subtract-the-Product-and-Sum-Of-Digits-of-an-Integer

class Solution {
    public int subtractProductAndSum(int n) {
        int pro = 1;
        int sum = 0;
        while(n!=0){
            int rem = n % 10;
            pro *= rem;
            sum += rem;
            n = n / 10;
        }
        int Result = pro - sum;
        return Result;
    }
}
